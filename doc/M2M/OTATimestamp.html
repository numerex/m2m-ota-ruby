def initialize(value,id = 0)
  case value
    when Hash
      super(value)
    when Time
      super(:id => id,:value => value.to_i * 1000)
    when Fixnum,Bignum
      super(:id => id,:value => value.to_i)
    else
      raise OTAException.new("Invalid timestamp value type: #{value.class}")
  end
end
