def self.from_w(buf)
  obj     = buf.unpack('CCn')
  objId   = obj[0]
  objType = obj[1]
  strLen  = obj[2]
  raise OTA_Exception.new("Invalid object from wire") if objType != OBJTYPE_STRING

  objValue = buf[4, strLen].unpack("A*")[0]
  OTA_String.new(objValue, objId)
end
