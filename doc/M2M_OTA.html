def crc(data)
  _m = 0
  _p = 0
  _r = data[0].unpack('C')[0]

  indexes = (1..data.length-1)
  indexes.each do |i|
    _d = data[i].unpack('C')[0]
    _r = (_r << 8) | _d
    _p = 0x0107 << 7
    _m = 0x8000
    while _m != 0x0080 do 
      if (_r & _m) != 0 then
        _r ^= _p
      end
      _p = (_p & 0x0000ffff) >> 1
      _m = (_m & 0x0000ffff) >> 1
    end
  end
  _r
end
